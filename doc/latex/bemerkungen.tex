\section{Zusätzliche Bemerkungen}
\begin{itemize}
    \item Für $e = 0$ (Kreisbahn):
    \[
    \mathbf{p}(\phi) = \frac{m \sqrt{GM}}{\sqrt{a}} \hat{\phi}
    \]
    
    \item Betrag des Impulses:
    \[
    |\mathbf{p}(\phi)| = \frac{m \sqrt{GM}}{\sqrt{a (1 - e^2)}} \sqrt{e^2 \sin^2 \phi + (1 + e \cos \phi)^2}
    \]
\end{itemize}
