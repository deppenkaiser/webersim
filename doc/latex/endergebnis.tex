\[
\boxed{ \mathbf{p}(\phi) = \frac{e m \sqrt{GM} \sin \phi}{\sqrt{a (1 - e^2)}} \hat{r} + \frac{m \sqrt{GM} (1 + e \cos \phi)}{\sqrt{a (1 - e^2)}} \hat{\phi} }
\]

\[
\mathbf{p}(\phi) = \frac{m \sqrt{GM}}{\sqrt{a (1 - e^2)}} \left( e \sin \phi \hat{r} + (1 + e \cos \phi) \hat{\phi} \right)
