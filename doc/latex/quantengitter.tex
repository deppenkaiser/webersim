\subsection{Knotenenergie aus Jones-Polynomen}
\[
E[V(t)] = \hbar c \cdot \oint_{|t|=1} \frac{V'(t)}{V(t)} \, dt
\]
\textbf{Beispiel (Quark)}: $V(t) = t + t^{-1} + t^{-2} \Rightarrow E \approx 3\hbar c/L_p$
