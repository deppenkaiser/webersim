\section{Gesamtstörungen auf Körper j}

\defbox{
\subsection*{Gesamtpositionsstörung}
Summe über alle Störkörper (i $\neq$ j):
\[
\Delta \mathbf{r}_j = \sum_{\substack{i=1 \\ i \neq j}}^N \Delta \mathbf{r}_{ij}
\]
}

\defbox{
\subsection*{Gesamtgeschwindigkeitsstörung}
\[
\Delta \mathbf{v}_j = \sum_{\substack{i=1 \\ i \neq j}}^N \Delta \mathbf{v}_{ij}
\]
}

\defbox{
\subsection*{Gesamtwinkelgeschwindigkeitsstörung}
\[
\Delta \omega_j = \sum_{\substack{i=1 \\ i \neq j}}^N \Delta \omega_{ij}
\]
}
