\begin{equation}
\Delta E_{\text{self}}^{\text{QED}} = \frac{\alpha \hbar c}{\pi} \int_0^{\infty} \frac{dk}{k} \cdot \mathcal{F}(k). \tag{7.2}

\begin{equation}
